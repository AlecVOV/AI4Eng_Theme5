# Create Bucket:
$AWS_REGION="us-east-1"

# Raw uploads
aws s3api create-bucket `
  --bucket 5g-dashboard-raw-data `
  --region $AWS_REGION

{
    "Location": "/5g-dashboard-raw-data",
    "BucketArn": "arn:aws:s3:::5g-dashboard-raw-data"
}

# Cleaned / feature store
aws s3api create-bucket `
  --bucket 5g-dashboard-cleaned-data `
  --region $AWS_REGION

{
    "Location": "/5g-dashboard-cleaned-data",
    "BucketArn": "arn:aws:s3:::5g-dashboard-cleaned-data"
}

# Model artifacts + inference CSVs
aws s3api create-bucket `
  --bucket 5g-dashboard-artifacts `
  --region $AWS_REGION

{
    "Location": "/5g-dashboard-artifacts",
    "BucketArn": "arn:aws:s3:::5g-dashboard-artifacts"
}


Add Policy (Block Public Access):
$buckets = @(
    "5g-dashboard-raw-data", 
    "5g-dashboard-cleaned-data", 
    "5g-dashboard-artifacts"
)

foreach ($bucket in $buckets) {
    aws s3api put-public-access-block `
      --bucket $bucket `
      --public-access-block-configuration BlockPublicAcls=true,IgnorePublicAcls=true,BlockPublicPolicy=true,RestrictPublicBuckets=true
}
